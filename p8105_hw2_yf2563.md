``` r
path_to_data = './data/Trash-Wheel-Collection-Totals-8-6-19.xlsx'
trashwheel_df = 
    read_xlsx(
        path = path_to_data,
        sheet = "Mr. Trash Wheel",
        range = cell_cols("A:N")) %>% 
    janitor::clean_names() %>% 
    drop_na(dumpster) %>% 
    mutate(
        sports_balls = round(sports_balls, digits = 0),
```
